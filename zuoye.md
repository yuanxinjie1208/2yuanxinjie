{    
    const string expr = "a()a(aaa(aaa(a)))aa()";    
    const char placehold = '#';    
    int num = 0;      
    stack<char> stk;    
    for (auto c : expr)    
    {        
        stk.push(c);        
        if (c == '(')            
                ++num;        
        if (num && c == ')')
        {            
            while (stk.top() != '(')                
                   stk.pop();            
            stk.pop();            
            stk.push(placehold);            
            --num;        
        }    
    }    
    string s;    
    while(!stk.empty())    
    {        
        s.insert(s.begin(),stk.top());        
        stk.pop();    
    }
